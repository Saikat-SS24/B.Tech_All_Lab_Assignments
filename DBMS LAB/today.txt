CREATE OR REPLACE FUNCTION SOMNATH2
RETURN NUMBER IS TOTAL NUMBER(2);
BEGIN
SELECT COUNT(*) INTO TOTAL FROM EMP WHERE GENDER='FEMALE';
RETURN TOTAL;
END;

SET SERVEROUTPUT ON;
DECLARE
K NUMBER(2);
BEGIN
K := SOMNATH2();
dbms_output.put_line('no of female'||K);
END;
